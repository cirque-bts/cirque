CREATE TABLE stmt_test (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(64) NOT NULL,
    datein DATETIME NOT NULL
) ENGINE=InnoDB, DEFAULT CHARACTER SET 'utf8';

;

DELIMITER $$

CREATE TRIGGER stmt_test_datein
    BEFORE INSERT ON stmt_test
    FOR EACH ROW BEGIN
        SET NEW.datein = NOW();
    END
$$

DELIMITER ;
